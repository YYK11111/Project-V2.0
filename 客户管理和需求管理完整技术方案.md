### 2.3 服务层设计

```typescript
// customer.service.ts
@Injectable()
export class CustomerService extends BaseService<Customer, CreateCustomerDto> {
  constructor(
    @InjectRepository(Customer)
    repository: Repository<Customer>,
    private interactionService: CustomerInteractionService,
  ) {
    super(Customer, repository);
  }

  async list(query: QueryCustomerDto) {
    const { name, level, status, industry, salesId } = query;
    return this.listBy({
      where: {
        name: this.sqlLike(name),
        level,
        status,
        industry,
        salesId,
      },
      relations: [],
    }, query);
  }

  async getCustomerDetail(id: string) {
    const customer = await this.detail(id);
    const interactions = await this.interactionService.findByCustomerId(id);
    return { ...customer, interactions };
  }

  async addInteraction(customerId: string, data: CreateInteractionDto) {
    return this.interactionService.create({
      ...data,
      customerId,
    });
  }

  async getCustomerStats(salesId?: string) {
    const where = salesId ? { salesId } : {};
    const [total, vip, important, regular, active] = await Promise.all([
      this.count({ where }),
      this.count({ where: { ...where, level: '1' } }),
      this.count({ where: { ...where, level: '2' } }),
      this.count({ where: { ...where, level: '3' } }),
      this.count({ where: { ...where, status: '3' } }),
    ]);
    return { total, vip, important, regular, active };
  }
}
```

### 3.3 服务层设计

```typescript
// requirement.service.ts
@Injectable()
export class RequirementService extends BaseService<Requirement, CreateRequirementDto> {
  constructor(
    @InjectRepository(Requirement)
    repository: Repository<Requirement>,
    private workflowEngine: WorkflowEngineService,
    private changeService: RequirementChangeService,
  ) {
    super(Requirement, repository);
  }

  async submit(data: CreateRequirementDto, userId: string) {
    // 1. 创建需求
    const requirement = this.repository.create({
      ...data,
      submitterId: userId,
      status: '1', // 待评审
      approvalStatus: '0', // 未提交
    });
    const saved = await this.repository.save(requirement);

    // 2. 启动审批流程
    const instance = await this.workflowEngine.startWorkflow(
      'requirement-approval',
      saved.id,
      {
        title: saved.title,
        type: saved.type,
        priority: saved.priority,
        submitterId: userId,
      },
      userId
    );

    // 3. 更新需求状态
    saved.workflowInstanceId = instance.id;
    saved.approvalStatus = '1'; // 审批中
    await this.repository.save(saved);

    return saved;
  }

  async approve(taskId: string, userId: string, action: 'approve' | 'reject', comment: string) {
    const instance = await this.workflowEngine.completeTask(taskId, userId, action, comment);
    
    const requirement = await this.repository.findOne({ where: { id: instance.businessKey } });
    
    if (action === 'approve' && instance.status === '2') {
      requirement.approvalStatus = '2'; // 已通过
      requirement.status = '2'; // 已评审
    } else if (action === 'reject') {
      requirement.approvalStatus = '3'; // 已拒绝
      requirement.status = '7'; // 已拒绝
    }
    
    await this.repository.save(requirement);
    return requirement;
  }

  async schedule(requirementId: string, assigneeId: string, estimatedWorkload: string) {
    const requirement = await this.repository.findOne({ where: { id: requirementId } });
    requirement.assigneeId = assigneeId;
    requirement.estimatedWorkload = estimatedWorkload;
    requirement.status = '3'; // 已排期
    return this.repository.save(requirement);
  }

  async createChange(data: CreateRequirementChangeDto, userId: string) {
    // 创建变更记录
    const change = await this.changeService.create({
      ...data,
      applicantId: userId,
      approvalStatus: '1',
    });

    // 启动变更审批流程
    const instance = await this.workflowEngine.startWorkflow(
      'requirement-change-approval',
      change.id,
      {
        requirementId: data.requirementId,
        changeType: data.changeType,
        description: data.changeDescription,
      },
      userId
    );

    change.workflowInstanceId = instance.id;
    await this.changeService.save(change);

    return change;
  }

  async getMyTasks(userId: string) {
    return this.workflowEngine.getPendingTasks(userId);
  }

  async list(query: QueryRequirementDto) {
    const { title, type, priority, status, customerId, projectId } = query;
    return this.listBy({
      where: {
        title: this.sqlLike(title),
        type,
        priority,
        status,
        customerId,
        projectId,
      },
      relations: ['customer', 'project', 'submitter', 'assignee'],
    }, query);
  }
}
```
