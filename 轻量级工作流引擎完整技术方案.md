### 1.2 模块划分

| 流程定义模块 | 管理流程模板 | 创建、编辑、发布流程定义 |

## 二、节点类型设计

### 2.1 支持的节点类型

| 节点类型 | 类型标识 | 说明 |
|---------|---------|------|
| 开始节点 | start | 流程起点，每个流程有且仅有一个 |
| 结束节点 | end | 流程终点，每个流程至少有一个 |
| 审批节点 | approval | 需要人工审批的节点 |
| 条件节点 | condition | 根据条件判断流转方向 |
| 通知节点 | notification | 发送系统/邮件/短信通知 |
| 抄送节点 | cc | 抄送给他人知会 |
| 延时节点 | delay | 延时一段时间后继续 |
| 表单节点 | form | 需要填写表单的节点 |

## 四、工作流引擎核心服务

### 4.1 引擎服务完整代码

```typescript
// workflow-engine.service.ts
import { Injectable, BadRequestException, ForbiddenException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { WorkflowDefinition } from './entity/workflow-definition.entity';
import { WorkflowInstance } from './entity/workflow-instance.entity';
import { WorkflowTask } from './entity/workflow-task.entity';
import { WorkflowHistory } from './entity/workflow-history.entity';
import { NodeHandlerFactory } from './handler/node-handler.factory';
import { NotificationService } from '../notices/service';
import { UserService } from '../users/users.service';
import { NodeConfig, NodeExecutionContext, NodeResult } from './interface/node.interface';

@Injectable()
export class WorkflowEngineService {
  constructor(
    @InjectRepository(WorkflowDefinition)
    private definitionRepo: Repository<WorkflowDefinition>,
    @InjectRepository(WorkflowInstance)
    private instanceRepo: Repository<WorkflowInstance>,
    @InjectRepository(WorkflowTask)
    private taskRepo: Repository<WorkflowTask>,
    @InjectRepository(WorkflowHistory)
    private historyRepo: Repository<WorkflowHistory>,
    private nodeHandlerFactory: NodeHandlerFactory,
    private notificationService: NotificationService,
    private userService: UserService,
  ) {}

  /**
   * 启动工作流
   * @param definitionCode 流程定义编码
   * @param businessKey 业务数据ID
   * @param variables 流程变量
   * @param starterId 发起人ID
   */
  async startWorkflow(
    definitionCode: string,
    businessKey: string,
    variables: Record<string, any>,
    starterId: string
  ): Promise<WorkflowInstance> {
    // 查询流程定义
    const definition = await this.definitionRepo.findOne({
      where: { code: definitionCode, isActive: '1' }
    });

    if (!definition) {
      throw new BadRequestException('流程定义不存在');
    }

    // 创建流程实例
    const instance = this.instanceRepo.create({
      definitionId: definition.id,
      definitionCode: definition.code,
      businessKey,
      starterId,
      variables,
      status: '1', // 进行中
      startTime: new Date().toISOString(),
    });

    const savedInstance = await this.instanceRepo.save(instance);

    // 执行第一个节点
    await this.executeNode(savedInstance, definition, 0);

    return savedInstance;
  }

  /**
   * 执行节点
   * @param instance 流程实例
   * @param definition 流程定义
   * @param nodeIndex 节点索引
   */
  private async executeNode(
    instance: WorkflowInstance,
    definition: WorkflowDefinition,
    nodeIndex: number
  ) {
    const node = definition.nodes[nodeIndex];
    const handler = this.nodeHandlerFactory.getHandler(node.type);

    const context: NodeExecutionContext = {
      instanceId: instance.id,
      nodeId: node.id,
      variables: instance.variables,
      startTime: new Date(),
    };

    try {
      // 1. 进入节点
      await handler.onEnter?.(context);

      // 2. 执行节点
      const result = await handler.execute(context);

      // 3. 记录历史
      await this.recordHistory(instance, node, context, 'execute');

      // 4. 处理结果，推进流程
      if (result.success && result.nextNodeIds?.length) {
        for (const nextNodeId of result.nextNodeIds) {
          const nextNode = definition.nodes.find(n => n.id === nextNodeId);
          if (nextNode) {
            const nextIndex = definition.nodes.indexOf(nextNode);
            await this.executeNode(instance, definition, nextIndex);
          }
        }
      }

      // 5. 退出节点
      await handler.onExit?.(context);

    } catch (error) {
      await this.handleNodeError(instance, node, error);
    }
  }

  /**
   * 创建审批任务
   */
  async createApprovalTask(
    instance: WorkflowInstance,
    node: NodeConfig,
    context: NodeExecutionContext
  ) {
    // 确定审批人
    let assigneeIds = await this.determineAssignees(node, context);

    if (node.properties.multiInstance) {
      // 会签任务
      const tasks = assigneeIds.map(userId => {
        return this.taskRepo.create({
          instanceId: instance.id,
          nodeId: node.id,
          nodeName: node.name,
          nodeType: node.type,
          assigneeId: userId,
          status: '1',
          startTime: new Date().toISOString(),
        });
      });
      await this.taskRepo.save(tasks);
    } else {
      // 单人任务
      const task = this.taskRepo.create({
        instanceId: instance.id,
        nodeId: node.id,
        nodeName: node.name,
        nodeType: node.type,
        assigneeId: assigneeIds[0],
        status: '1',
        startTime: new Date().toISOString(),
      });
      await this.taskRepo.save(task);
    }

    // 发送通知
    await this.notificationService.send({
      receivers: assigneeIds,
      title: '待办审批通知',
      content: `您有一个待审批任务：${node.name}`,
      type: 'system',
    });
  }

  /**
   * 完成任务
   */
  async completeTask(
    taskId: string,
    userId: string,
    action: 'approve' | 'reject',
    comment?: string
  ): Promise<WorkflowInstance> {
    const task = await this.taskRepo.findOne({
      where: { id: taskId },
      relations: ['instance']
    });

    if (!task || task.status !== '1') {
      throw new BadRequestException('任务不存在或已处理');
    }

    if (task.assigneeId !== userId) {
      throw new ForbiddenException('您不是该任务的审批人');
    }

    const instance = task.instance;
    const definition = await this.definitionRepo.findOne({
      where: { id: instance.definitionId }
    });

    // 更新任务状态
    task.status = action === 'approve' ? '3' : '4';
    task.action = action;
    task.comment = comment;
    task.completeTime = new Date().toISOString();
    await this.taskRepo.save(task);

    // 记录历史
    await this.recordHistory(instance, null, { ...task, variables: instance.variables }, action);

    // 推进流程
    if (action === 'approve') {
      const currentNode = definition.nodes.find(n => n.id === task.nodeId);
      const nextNode = this.findNextNode(definition, currentNode, instance.variables);
      if (nextNode) {
        await this.executeNode(instance, definition, definition.nodes.indexOf(nextNode));
      } else {
        await this.completeInstance(instance);
      }
    } else {
      await this.rejectInstance(instance);
    }

    return instance;
  }

  /**
   * 获取用户待办任务
   */
  async getPendingTasks(userId: string): Promise<WorkflowTask[]> {
    return this.taskRepo.find({
      where: { assigneeId: userId, status: '1' },
      relations: ['instance'],
      order: { createTime: 'DESC' }
    });
  }

  /**
   * 转交任务
   */
  async transferTask(
    taskId: string,
    userId: string,
    targetUserId: string,
    comment?: string
  ): Promise<WorkflowTask> {
    const task = await this.taskRepo.findOne({ where: { id: taskId } });

    if (!task || task.status !== '1') {
      throw new BadRequestException('任务不存在或已处理');
    }

    if (task.assigneeId !== userId) {
      throw new ForbiddenException('您不是该任务的审批人');
    }

    // 记录原审批人
    const originalAssignee = task.assigneeId;

    // 更新任务
    task.assigneeId = targetUserId;
    await this.taskRepo.save(task);

    // 发送通知给新的审批人
    await this.notificationService.send({
      receivers: [targetUserId],
      title: '任务转交通知',
      content: `您被分配了一个审批任务，请及时处理`,
      type: 'system',
    });

    // 记录历史
    await this.recordHistory(task.instance, null, {
      nodeId: task.nodeId,
      nodeName: task.nodeName,
      operatorId: userId,
      action: 'transfer',
      comment: `转交给${targetUserId}`,
    }, 'transfer');

    return task;
  }

  /**
   * 催办任务
   */
  async remindTask(taskId: string, userId: string) {
    const task = await this.taskRepo.findOne({
      where: { id: taskId },
      relations: ['instance']
    });

    if (!task || task.status !== '1') {
      throw new BadRequestException('任务不存在或已处理');
    }

    // 发送催办通知
    await this.notificationService.send({
      receivers: [task.assigneeId],
      title: '任务催办通知',
      content: `您有一个待审批任务被催办，请尽快处理`,
      type: 'system',
    });

    return { success: true };
  }

  /**
   * 确定审批人
   */
  private async determineAssignees(node: NodeConfig, context: NodeExecutionContext): Promise<string[]> {
    const { assigneeType, assigneeValue } = node.properties;

    switch (assigneeType) {
      case 'user':
        return [assigneeValue as string];
      case 'role':
        return this.userService.findUserIdsByRoleId(assigneeValue as string);
      case 'department':
        return this.userService.findUserIdsByDeptId(assigneeValue as string);
      case 'dynamic':
        return this.evaluateExpression(assigneeValue as string, context.variables);
      default:
        return [];
    }
  }

  /**
   * 查找下一节点
   */
  private findNextNode(definition: WorkflowDefinition, currentNode: NodeConfig, variables: Record<string, any>): NodeConfig | null {
    const outgoingFlows = definition.flows.filter(f => f.sourceNodeId === currentNode.id);
    if (outgoingFlows.length === 0) return null;

    // 查找有条件的分支
    const conditionalFlow = outgoingFlows.find(f => {
      if (!f.condition) return false;
      return this.evaluateCondition(f.condition, variables);
    });

    if (conditionalFlow) {
      return definition.nodes.find(n => n.id === conditionalFlow.targetNodeId);
    }

    // 默认取第一个
    return definition.nodes.find(n => n.id === outgoingFlows[0].targetNodeId);
  }

  /**
   * 评估条件
   */
  private evaluateCondition(condition: any, variables: Record<string, any>): boolean {
    const fieldValue = this.getFieldValue(condition.field, variables);
    switch (condition.operator) {
      case 'eq': return fieldValue === condition.value;
      case 'neq': return fieldValue !== condition.value;
      case 'gt': return Number(fieldValue) > Number(condition.value);
      case 'gte': return Number(fieldValue) >= Number(condition.value);
      case 'lt': return Number(fieldValue) < Number(condition.value);
      case 'lte': return Number(fieldValue) <= Number(condition.value);
      case 'in': return Array.isArray(condition.value) && condition.value.includes(fieldValue);
      case 'contains': return String(fieldValue).includes(String(condition.value));
      default: return false;
    }
  }

  /**
   * 获取字段值
   */
  private getFieldValue(field: string, variables: Record<string, any>): any {
    if (field.startsWith('${') && field.endsWith('}')) {
      const varName = field.slice(2, -1);
      if (varName.startsWith('variables.')) {
        return variables[varName.slice(10)];
      }
      return variables[varName];
    }
    return variables[field];
  }

  /**
   * 解析表达式
   */
  private evaluateExpression(expr: string, variables: Record<string, any>): string[] {
    if (expr.startsWith('${') && expr.endsWith('}')) {
      const varName = expr.slice(2, -1);
      const value = this.getFieldValue(varName, variables);
      return Array.isArray(value) ? value : [value];
    }
    return [expr];
  }

  /**
   * 完成流程实例
   */
  private async completeInstance(instance: WorkflowInstance) {
    instance.status = '2'; // 已完成
    instance.endTime = new Date().toISOString();
    instance.duration = new Date(instance.endTime).getTime() - new Date(instance.startTime).getTime();
    await this.instanceRepo.save(instance);
  }

  /**
   * 驳回流程实例
   */
  private async rejectInstance(instance: WorkflowInstance) {
    instance.status = '3'; // 已拒绝
    instance.endTime = new Date().toISOString();
    await this.instanceRepo.save(instance);
  }

  /**
   * 记录历史
   */
  private async recordHistory(instance: WorkflowInstance, node: NodeConfig | null, context: any, action: string) {
    const history = this.historyRepo.create({
      instanceId: instance.id,
      taskId: context.id,
      nodeId: node?.id || context.nodeId,
      nodeName: node?.name || context.nodeName,
      operatorId: context.operatorId,
      action,
      comment: context.comment,
      variables: context.variables,
    });
    await this.historyRepo.save(history);
  }

  /**
   * 处理节点错误
   */
  private async handleNodeError(instance: WorkflowInstance, node: NodeConfig, error: Error) {
    console.error(`[Workflow Error] Node ${node.id}: ${error.message}`);
    instance.status = '3';
    instance.endTime = new Date().toISOString();
    await this.instanceRepo.save(instance);
    throw error;
  }
}
```

```typescript
// 业务集成示例：需求审批
@Injectable()
export class RequirementService {
  constructor(
    private workflowEngine: WorkflowEngineService,
    private requirementRepo: Repository<Requirement>,
  ) {}

  async submit(data: CreateRequirementDto, userId: string) {
    // 1. 保存需求
    const requirement = this.requirementRepo.create({
      ...data,
      submitterId: userId,
      status: 'submitted',
    });
    const saved = await this.requirementRepo.save(requirement);

    // 2. 启动审批流程
    await this.workflowEngine.startWorkflow(
      'requirement-approval',
      saved.id,
      { title: data.title, description: data.description, priority: data.priority, submitterId: userId },
      userId
    );

    return saved;
  }

  async approve(taskId: string, userId: string, action: 'approve' | 'reject', comment: string) {
    const instance = await this.workflowEngine.completeTask(taskId, userId, action, comment);
    const requirement = await this.requirementRepo.findOne({ where: { id: instance.businessKey } });

    if (action === 'approve' && instance.status === '2') {
      requirement.status = 'approved';
    } else if (action === 'reject') {
      requirement.status = 'rejected';
    }

    await this.requirementRepo.save(requirement);
    return requirement;
  }

  async getMyTasks(userId: string) {
    return this.workflowEngine.getPendingTasks(userId);
  }
}
```
